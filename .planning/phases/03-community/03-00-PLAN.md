    Part 4 — update_trending_posts() function (D-06). MUST be plain LANGUAGE plpgsql WITHOUT SECURITY DEFINER — pg_cron runs as postgres which has BYPASSRLS (RESEARCH §Security Domain warning):
    ```sql
    -- ─── Part 4: update_trending_posts (D-06) ───
    CREATE OR REPLACE FUNCTION public.update_trending_posts()
    RETURNS void
    LANGUAGE plpgsql
    AS $$
    DECLARE
      v_cutoff TIMESTAMPTZ := NOW() - INTERVAL '24 hours';
    BEGIN
      -- Clear existing trending flags
      UPDATE public.community_posts
        SET is_trending = FALSE
        WHERE is_trending = TRUE;

      -- Flag top 5 by trending score within 24h window
      WITH scored AS (
        SELECT
          id,
          (like_count * 2 + comment_count * 3 + view_count * 0.1) *
            (1 + GREATEST(0, 1 - EXTRACT(EPOCH FROM (NOW() - created_at)) / 86400)) AS score
        FROM public.community_posts
        WHERE created_at >= v_cutoff
          AND is_blinded = FALSE
        ORDER BY score DESC
        LIMIT 5
      )
      UPDATE public.community_posts
        SET is_trending = TRUE
        WHERE id IN (SELECT id FROM scored);
    END;
    $$;
    ```

        PERFORM cron.unschedule('update-trending-posts');
      END IF;
    END $$;

    SELECT cron.schedule(
      'update-trending-posts',
      '*/10 * * * *',
