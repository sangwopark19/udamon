    if (!user) return;

    if (updates.nickname && updates.nickname !== user.nickname) {
      if (user.nickname_changed_at) {
        const lastChanged = new Date(user.nickname_changed_at);
        const daysSince = (Date.now() - lastChanged.getTime()) / (1000 * 60 * 60 * 24);
