**Step 1-A — StudioScreen state machine 재구성:**

```typescript
// app/src/screens/photographer/StudioScreen.tsx 재작성 핵심 구조
import * as photographerApi from '../../services/photographerApi';
import { usePhotographer } from '../../contexts/PhotographerContext';
import { useAuth } from '../../contexts/AuthContext';
import GradeBadge from '../../components/photographer/GradeBadge';
import type { PhotographerApplication } from '../../types/photographerApplication';
import type { Photographer } from '../../types/photographer';

type StudioState =
  | { kind: 'loading' }
  | { kind: 'no_application' }
  | { kind: 'pending'; application: PhotographerApplication }
  | { kind: 'approved'; photographer: Photographer }
  | { kind: 'rejected'; application: PhotographerApplication };

export default function StudioScreen() {
  const { t } = useTranslation();
  const navigation = useNavigation<Nav>();
  const insets = useSafeAreaInsets();
  const { user } = useAuth();
  const { photoPosts, getPhotoPostsByPhotographer, getPhotographer } = usePhotographer();
  const [state, setState] = useState<StudioState>({ kind: 'loading' });

  const loadState = useCallback(async () => {
    if (!user?.id) {
      setState({ kind: 'no_application' });
      return;
    }
    setState({ kind: 'loading' });
    const appRes = await photographerApi.fetchMyPhotographerApplication(user.id);
    if (appRes.error) {
      // 에러 시 토스트 + no_application 으로 fallback (사용자가 신청 할 수 있도록)
      console.warn('[Studio] fetchMyPhotographerApplication error', appRes.error);
      setState({ kind: 'no_application' });
      return;
    }
    const app = appRes.data;
    if (!app) {
      setState({ kind: 'no_application' });
      return;
    }
    if (app.status === 'pending') {
      setState({ kind: 'pending', application: app });
      return;
    }
    if (app.status === 'rejected') {
      setState({ kind: 'rejected', application: app });
      return;
    }
    // approved — photographer row 조회
    const pgRes = await photographerApi.fetchPhotographerByUserId(user.id);
    if (pgRes.data) {
      setState({ kind: 'approved', photographer: pgRes.data });
    } else {
      // 트리거가 아직 반영 안 된 edge case (race) — pending 으로 표시
      setState({ kind: 'pending', application: app });
    }
  }, [user?.id]);

  useEffect(() => {
    void loadState();
  }, [loadState]);

  // ─── Render ─────────────────────────────
  if (state.kind === 'loading') {
    return <StudioLoading insets={insets} />;
  }
  if (state.kind === 'no_application') {
    return (
      <StudioEmptyHero
        insets={insets}
        iconName="camera-outline"
        iconColor={colors.textTertiary}
        title={t('studio_signup_desc')}
        desc={t('studio_signup_desc')}
        ctaLabel={t('studio_signup_cta')}
        onCta={() => navigation.navigate('PhotographerRegister')}
      />
    );
  }
  if (state.kind === 'pending') {
    return (
      <StudioEmptyHero
        insets={insets}
        iconName="time-outline"
        iconColor={colors.warning}
        iconSize={64}
        title={t('studio_pending_title')}
        desc={t('studio_pending_desc')}
        hint={t('studio_pending_notification_hint')}
      />
    );
  }
  if (state.kind === 'rejected') {
    return (
      <StudioRejectedHero
        insets={insets}
        application={state.application}
        onReapply={() => navigation.navigate('PhotographerRegister')}
      />
    );
  }
  // approved — 기존 Studio UI + 상단 GradeBadge
  const myPosts = getPhotoPostsByPhotographer(state.photographer.id);
  return (
    <StudioApprovedLayout
      insets={insets}
      photographer={state.photographer}
      posts={myPosts}
      navigation={navigation}
    />
  );
}
```

// app/src/navigation/MainTabNavigator.tsx 수정
import { useAuth } from '../contexts/AuthContext';
import * as photographerApi from '../services/photographerApi';

export default function MainTabNavigator() {
  const { user } = useAuth();
  const [studioState, setStudioState] = useState<'no_app' | 'pending' | 'approved' | 'rejected'>('no_app');

  useEffect(() => {
    if (!user?.id) { setStudioState('no_app'); return; }
    let cancelled = false;
    photographerApi.fetchMyPhotographerApplication(user.id).then((res) => {
      if (cancelled) return;
      if (!res.data) setStudioState('no_app');
      else if (res.data.status === 'pending') setStudioState('pending');
      else if (res.data.status === 'rejected') setStudioState('rejected');
      else setStudioState('approved');
    });
    return () => { cancelled = true; };
  }, [user?.id]);

  const studioLabel =
    studioState === 'pending' ? t('tab_pending_review') :
    studioState === 'approved' ? t('tab_studio') :
    t('tab_photographer');
  const studioIconName: keyof typeof Ionicons.glyphMap =
    studioState === 'pending' ? 'time-outline' :
    studioState === 'approved' ? 'aperture-outline' :
    'camera-outline';
  const studioIconActive: keyof typeof Ionicons.glyphMap =
    studioState === 'pending' ? 'time' :
    studioState === 'approved' ? 'aperture' :
    'camera';

  return (
    <Tab.Navigator ...>
      {/* ... existing tabs ... */}
      <Tab.Screen
        name="Studio"
        component={StudioScreen}
        options={{
          tabBarLabel: studioLabel,
          tabBarIcon: ({ focused, color, size }) => (
            <Ionicons name={focused ? studioIconActive : studioIconName} size={size} color={color} />
          ),
        }}
      />
      {/* ... */}
    </Tab.Navigator>
  );
}

export default function CollectionDetailScreen() {
  const route = useRoute<Route>();
  const collectionId = route.params.collectionId;
  const { getCollectionPosts, collections } = usePhotographer();

  const [posts, setPosts] = useState<PhotoPost[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const load = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await getCollectionPosts(collectionId);
      setPosts(result);
    } catch (e: unknown) {
      setError(e instanceof Error ? e.message : 'Failed to load');
    } finally {
      setLoading(false);
    }
  }, [collectionId, getCollectionPosts]);

  useEffect(() => {
    void load();
  }, [load]);

  if (loading) return <Skeleton type="grid" />;
  if (error) return <EmptyState icon="cloud-offline-outline" title={t('pg_feed_loading_failed')} ctaLabel={t('pg_feed_loading_retry')} onCta={load} />;
  // ... 기존 posts 렌더 로직 유지 (FlatList grid)
